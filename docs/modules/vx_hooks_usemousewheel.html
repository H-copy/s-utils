 export default {
   setup(){
     const {
       isUp,
       isDown,
       directionY,
       wheelEvent,
       unUse,
       canUse,
       onMousewheel,
     }
     const count = ref(0)

     watch(wheelEvent, () => { *
       if(isUp(directionY.value)){
         count.value -= 1
       } *
       if(isDown(directionY.value)){
         count.value += 1
       }

     })

     return {
       onMousewheel,
       unUse,
       canUse,
     }
   }
 }
